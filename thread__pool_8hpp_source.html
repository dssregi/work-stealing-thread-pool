215inline void ThreadPool::worker(std::stop_token token, int idx) {
216 TaskFunc task;
217
218 while (!token.stop_requested()) {
219 // 1. Primary: Try LIFO pop from own queue (optimal cache use)
220 if (work_queues[idx].try_pop(task)) {
221 task();
222 continue;
223 }
224
225 // 2. Stealing: Check a random queue
226 int i = get_random();
227
228 // Use try_steal (FIFO pop) from the random queue
229 if (work_queues[i].try_steal(task)) {
230 task();
231 continue;
232 }
233
234 // 3. Last Resort: Block efficiently on our own queue (LIFO pop)
235 // If wait_and_pop returns false, it means close() was called and the queue is empty.
236 if (!work_queues[idx].wait_and_pop(task)) {
237 break;
238 }
239
240 if (task) {
241 task();
242 }
243 }
244 std::cout << "Worker " << idx << " exited." << std::endl;
245}
