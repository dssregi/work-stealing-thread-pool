171 void operator()() const {
172 // Loops over the assigned depth slice range (Z-axis)
173 for (int z = start_slice_; z < end_slice_; ++z) {
174 // Loops over rows (Y-axis) and columns (X-axis)
175 for (int r = BORDER; r < IMG_HEIGHT - BORDER; ++r) {
176 for (int c = BORDER; c < IMG_WIDTH - BORDER; ++c) {
177
178 float sum = 0.0f;
179 int kernel_idx = 0;
180
181 // Iterate over the 3D kernel window (kz, kr, kc)
182 for (int kz = -BORDER; kz <= BORDER; ++kz) {
183 for (int kr = -BORDER; kr <= BORDER; ++kr) {
184 for (int kc = -BORDER; kc <= BORDER; ++kc) {
185
186 int iz = z + kz;
187 int ir = r + kr;
188 int ic = c + kc;
189
190 int input_idx = get_index(iz, ir, ic);
191 float k_val = kernel_[kernel_idx++];
192
193 sum += input_[input_idx] * k_val;
194 }
195 }
196 }
197
198 // Write the calculated value to the output image
199 output_[get_index(z, r, c)] = sum;
200 }
201 }
202 }
203
204 // Signal completion using the atomic counter
205 completed_slices_counter_.fetch_add(end_slice_ - start_slice_);
206 }

265inline float calculate_std_dev(const Image& img, const std::string& label) {
266 // Sample a uniform region: the background in the first few slices (excluding borders)
267 constexpr int SAMPLE_Z_END = 5;
268 std::vector<float> sample;
269 int index_count = 0;
270
271 for (int z = BORDER; z < SAMPLE_Z_END; ++z) {
272 for (int r = BORDER; r < IMG_HEIGHT - BORDER; ++r) {
273 for (int c = BORDER; c < IMG_WIDTH - BORDER; ++c) {
274 // The cube starts at Z=5, so this captures only background noise
275 sample.push_back(img[z * IMG_WIDTH * IMG_HEIGHT + r * IMG_WIDTH + c]);
276 index_count++;
277 }
278 }
279 }
280
281 if (sample.empty()) return 0.0f;
282
283 // Calculate Mean
284 double sum = 0.0;
285 for (float val : sample) {
286 sum += val;
287 }
288 float mean = sum / index_count;
289
290 // Calculate Variance
291 double variance_sum = 0.0;
292 for (float val : sample) {
293 variance_sum += std::pow(val - mean, 2);
294 }
295 float variance = variance_sum / (index_count - 1); // Use (N-1) for sample standard deviation
296
297 // Calculate Standard Deviation
298 float std_dev = std::sqrt(variance);
299
300 std::cout << " " << label << " (Background region): Std Dev = " << std_dev << std::endl;
301 return std_dev;
302}
